// Mapping for letters to corresponding encoded values const letterMap = { 'A': 'B03', 'B': 'C05', 'C': 'D08', 'D': 'E10', 'E': 'F13', 'F': 'G15', 'Z': '077' }; // Function to encode user input (convert letters to their encoded values) function encodePasswordInput(input) { let encodedString = ''; for (let i = 0; i < input.length; i++) { let letter = input[i]; if (letterMap[letter]) { encodedString += letterMap[letter]; } else { // Handle invalid input (if any letter is not in letterMap) console.warn(`Invalid character: ${letter}`); } } return encodedString; }
